  402.     return function(apiPath, params, callback) {
  403.         var promise = new Promise(function(resolve, reject) {
  404.             requestMethod.call(this, apiPath, params, function(err, resp) {
  405.                 if (err) {

  427. /**
  428.  * Wraps a function for use as a request callback
  429.  *
  430.  * @param  {TumblrClient~callback} callback - function to wrap
  431.  *
  432.  * @return {TumblrClient~callback} request callback
  433.  *
  434.  * @private
  435.  */
  436. function requestCallback(callback) {
  437.     if (!callback) {
  438.         return undefined;
  439.     }
  441.     return function(err, response, body) {
  442.         if (err) {
  443.             return callback(err, null, response);
  444.         }
  446.         if (response.statusCode >= 400) {
  447.             var errString = body.meta ? body.meta.msg : body.error;
  448.             return callback(new Error('API error: ' + response.statusCode + ' ' + errString), null, response);
  449.         }
  451.         if (body && body.response) {
  452.             return callback(null, body.response, response);
  453.         } else {
  454.             return callback(new Error('API error (malformed API response): ' + body), null, response);
  455.         }

  488. /**
  489.  * Create a function to make POST requests to the Tumblr API
  490.  *
  491.  * @param  {Function} requestPost - function that performs a get request
  492.  * @param  {Object} credentials - OAuth credentials
  493.  * @param  {string} baseUrl - base URL for the request
  494.  * @param  {string} apiPath - URL path for the request
  495.  * @param  {Object} requestOptions - additional request options
  496.  * @param  {Object} params - form data
  497.  * @param  {TumblrClient~callback} callback - request callback
  498.  *
  499.  * @return {Request} Request object
  500.  *
  501.  * @private
  502.  */
  503. function postRequest(requestPost, credentials, baseUrl, apiPath, requestOptions, params, callback) {
  504.     params = params || {};
  506.     // Sign without multipart data
  507.     var currentRequest = requestPost(extend({
  508.         url: baseUrl + apiPath,
  509.         oauth: credentials,
  510.     }, requestOptions), function(err, response, body) {
  511.         try {
  512.             body = JSON.parse(body);
  513.         } catch (e) {
  514.             body = {
  515.                 error: 'Malformed Response: ' + body,
  516.             };
  517.         }
  518.         requestCallback(callback)(err, response, body);
  519.     });
  521.     // Sign it with the non-data parameters
  522.     var dataKeys = ['data'];
  523.     currentRequest.form(omit(params, dataKeys));
  524.     currentRequest.oauth(credentials);
  526.     // Clear the side effects from form(param)
  527.     delete currentRequest.headers['content-type'];
  528.     delete currentRequest.body;
  530.     // if 'data' is an array, rename it with indices
  531.     if ('data' in params && Array.isArray(params.data)) {
  532.         for (var i = 0; i < params.data.length; ++i) {
  533.             params['data[' + i + ']'] = params.data[i];
  534.         }
  535.         delete params.data;
  536.     }
  538.     // And then add the full body
  539.     var form = currentRequest.form();
  540.     for (var key in params) {
  541.         form.append(key, params[key]);
  542.     }
  544.     // Add the form header back
  545.     extend(currentRequest.headers, form.getHeaders());
  547.     return currentRequest;
  548. }
  550. /**
  551.  * Adds a request method to the client
  552.  *
  553.  * @param  {Object} client - add the method to this object
  554.  * @param  {string} methodName - the name of the method
  555.  * @param  {string} apiPath - the API route, which uses any colon-prefixed segments as arguments
  556.  * @param  {Array} paramNames - ordered list of required request parameters used as arguments
  557.  * @param  {String|Function} requestType - the request type or a function that makes the request
  558.  *
  559.  * @private
  560.  */
  561. function addMethod(client, methodName, apiPath, paramNames, requestType) {
  562.     var apiPathSplit = apiPath.split('/');
  563.     var apiPathParamsCount = apiPath.split(/\/:[^\/]+/).length - 1;
  565.     var buildApiPath = function(args) {
  566.         var pathParamIndex = 0;
  567.         return reduce(apiPathSplit, function(apiPath, apiPathChunk, i) {
  568.             // Parse arguments in the path
  569.             if (apiPathChunk === ':blogIdentifier') {
  570.                 // Blog URLs are special
  571.                 apiPathChunk = forceFullBlogUrl(args[pathParamIndex++]);
  572.             } else if (apiPathChunk[0] === ':') {
  573.                 apiPathChunk = args[pathParamIndex++];
  574.             }
  576.             if (apiPathChunk) {
  577.                 return apiPath + '/' + apiPathChunk;
  578.             } else {
  579.                 return apiPath;
  580.             }
  581.         }, '');
  582.     };
  584.     var namedParams = (apiPath.match(/\/:[^\/]+/g) || []).map(function(param) {
  585.         return param.substr(2);
  586.     }).concat(paramNames, 'params', 'callback');
  588.     var methodBody = function() {
  589.         var argsLength = arguments.length;
  590.         var args = new Array(argsLength);
  591.         for (var i = 0; i < argsLength; i++) {
  592.             args[i] = arguments[i];
  593.         }
  595.         var requiredParamsStart = apiPathParamsCount;
  596.         var requiredParamsEnd = requiredParamsStart + paramNames.length;
  597.         var requiredParamArgs = args.slice(requiredParamsStart, requiredParamsEnd);
  599.         // Callback is at the end
  600.         var callback = isFunction(args[args.length - 1]) ? args.pop() : null;
  602.         // Required Parmas
  603.         var params = zipObject(paramNames, requiredParamArgs);
  604.         extend(params, isPlainObject(args[args.length - 1]) ? args.pop() : {});
  606.         // Path arguments are determined after required parameters
  607.         var apiPathArgs = args.slice(0, apiPathParamsCount);
  609.         var request = requestType;
  610.         if (isString(requestType)) {
  611.             request = requestType.toUpperCase() === 'POST' ? client.postRequest : client.getRequest;
  612.         } else if (!isFunction(requestType)) {
  613.             request = client.getRequest;
  614.         }
  616.         return request.call(client, buildApiPath(apiPathArgs), params, callback);
  617.     };
  619.     set(client, methodName, createFunction(methodName, namedParams, methodBody));
  620. }

  637.             paramNames = [];
  638.         } else if (isPlainObject(apiPath)) {

  674.                 }
  675.                 return true;

  680.             if (!validate(params)) {
  681.                 throw new Error('Error validating parameters');
  682.             }
  683.         }

  686.             return this.createPost(blogIdentifier, params, callback);
